function Calendar({ selectedDate, onSelect }) {
  const [year, setYear] = useState(2025);
  const [month, setMonth] = useState(5);

  const changeMonth = dir => {
    let m = month + dir, y = year;
    if (m < 0) { m = 11; y--; }
    if (m > 11) { m = 0; y++; }
    setMonth(m); setYear(y);
  };

  const today = new Date();
  const firstDay = new Date(year, month, 1).getDay();
  const offset = firstDay === 0 ? 6 : firstDay - 1;
  const daysInMonth = new Date(year, month + 1, 0).getDate();

  return (
    <div className="calendar">
      <div className="cal-header">
        <button className="cal-nav" onClick={() => changeMonth(-1)}>‹</button>
        <span className="cal-month">{MONTHS[month]} {year}</span>
        <button className="cal-nav" onClick={() => changeMonth(1)}>›</button>
      </div>
      <div className="cal-days">
        {DAYS.map(d => <div key={d} className="cal-day-label">{d}</div>)}
        {Array.from({length: offset}).map((_, i) => <div key={`e${i}`} className="cal-day empty"></div>)}
        {Array.from({length: daysInMonth}).map((_, i) => {
          const d = i + 1;
          const thisDate = new Date(year, month, d);
          const past = thisDate < new Date(today.getFullYear(), today.getMonth(), today.getDate());
          const isSel = selectedDate && selectedDate.d === d && selectedDate.m === month && selectedDate.y === year;
          const isToday = d === today.getDate() && month === today.getMonth() && year === today.getFullYear();
          let cls = 'cal-day';
          if (past) cls += ' disabled';
          else if (isSel) cls += ' selected';
          else if (isToday) cls += ' today';
          return (
            <div key={d} className={cls} onClick={() => !past && onSelect({d, m:month, y:year})}>
              {d}
            </div>
          );
        })}
      </div>
    </div>
  );
}

function App() {
  const [page, setPage] = useState('home');
  const [filter, setFilter] = useState('all');
  const [selectedProvider, setSelectedProvider] = useState(null);

  const handleSetPage = useCallback((p) => {
    setPage(p);
    window.scrollTo(0, 0);
  }, []);

  const handleSetFilter = useCallback((f) => setFilter(f), []);

  const handleSetProvider = useCallback((p) => setSelectedProvider(p), []);

  return (
    <div>
      <Nav page={page} setPage={handleSetPage} setFilter={handleSetFilter} />
      {page === 'home' && (
        <HomePage setPage={handleSetPage} setFilter={handleSetFilter} setSelectedProvider={handleSetProvider} />
      )}
      {page === 'providers' && (

      {page === 'detail' && (

      )}
      {page === 'booking' && (
        <BookingPage provider={selectedProvider} setPage={handleSetPage} />
      )}
      <Footer setPage={handleSetPage} />
    </div>
  );
}
